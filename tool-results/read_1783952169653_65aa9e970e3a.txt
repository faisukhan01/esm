   185тЖТ   185тЖТfunction ClassAttendance({ user, cls, students }: { user: any; cls: ClassInfo; students: any[] }) {
   186тЖТ   186тЖТ  const [attendance, setAttendance] = useState<Record<string, string>>({});
   187тЖТ   187тЖТ  const [saving, setSaving] = useState(false);
   188тЖТ   188тЖТ
   189тЖТ   189тЖТ  const setStatus = (id: string, status: string) => setAttendance(a => ({ ...a, [id]: status }));
   190тЖТ   190тЖТ  const present = students.filter(s => (attendance[s.id] || 'Present') === 'Present').length;
   191тЖТ   191тЖТ  const absent = students.filter(s => attendance[s.id] === 'Absent').length;
   192тЖТ   192тЖТ  const late = students.filter(s => attendance[s.id] === 'Late').length;
   193тЖТ   193тЖТ
   194тЖТ   194тЖТ  const save = async () => {
   195тЖТ   195тЖТ    if (students.length === 0) { toast({ title: 'No students to mark', variant: 'destructive' }); return; }
   196тЖТ   196тЖТ    setSaving(true);
   197тЖТ   197тЖТ    try {
   198тЖТ   198тЖТ      const records = students.map(s => ({ studentId: s.id, status: attendance[s.id] || 'Present' }));
   199тЖТ   199тЖТ      await api.markAttendance({
   200тЖТ   200тЖТ        classId: cls.id,
   201тЖТ   201тЖТ        branchId: user.branchId,
   202тЖТ   202тЖТ        class: cls.name,
   203тЖТ   203тЖТ        date: new Date().toISOString().slice(0, 10),
   204тЖТ   204тЖТ        teacherId: user.id,
   205тЖТ   205тЖТ        records,
   206тЖТ   206тЖТ      });
   207тЖТ   207тЖТ      toast({ title: 'Attendance saved!', description: `${present} present, ${absent} absent, ${late} late` });

   214тЖТ   214тЖТ        toast({ title: 'Session expired', description: 'Please sign out and sign in again.', variant: 'destructive' });
   215тЖТ   215тЖТ      } else {
   216тЖТ   216тЖТ        toast({ title: 'Failed to save', description: msg, variant: 'destructive' });
   217тЖТ   217тЖТ      }
   218тЖТ   218тЖТ    } finally { setSaving(false); }
   219тЖТ   219тЖТ  };
   220тЖТ   220тЖТ
   221тЖТ   221тЖТ  if (students.length === 0) {
   222тЖТ   222тЖТ    return <EmptyState icon={Users} title="No students in this class" desc="Your Branch Manager needs to add students to this class before you can take attendance." />;

   231тЖТ   231тЖТ        </Button>

   235тЖТ   235тЖТ        <Card className="p-4 text-center"><XCircle className="h-6 w-6 text-rose-600 mx-auto mb-1" /><div className="text-2xl font-bold">{absent}</div><div className="text-xs text-muted-foreground">Absent</div></Card>

   239тЖТ   239тЖТ        <Table>
   240тЖТ   240тЖТ          <TableHeader><TableRow className="bg-muted/40 hover:bg-muted/40"><TableHead>Roll</TableHead><TableHead>Student</TableHead><TableHead className="text-right">Mark</TableHead></TableRow></TableHeader>

   268тЖТ   268тЖТfunction ClassResults({ user, cls, courseId, students }: { user: any; cls: ClassInfo; courseId: string; students: any[] }) {
   269тЖТ   269тЖТ  const [exam, setExam] = useState('');
   270тЖТ   270тЖТ  const [totalMarks, setTotalMarks] = useState(100);
   271тЖТ   271тЖТ  const [marks, setMarks] = useState<Record<string, string>>({});
   272тЖТ   272тЖТ  const [saving, setSaving] = useState(false);
   273тЖТ   273тЖТ
   274тЖТ   274тЖТ  const calcGrade = (m: number, total: number) => {
   275тЖТ   275тЖТ    const p = (m / total) * 100;
   276тЖТ   276тЖТ    return p >= 90 ? 'A+' : p >= 80 ? 'A' : p >= 70 ? 'B' : p >= 60 ? 'C' : p >= 50 ? 'D' : 'F';
   277тЖТ   277тЖТ  };
   278тЖТ   278тЖТ
   279тЖТ   279тЖТ  const save = async () => {
   280тЖТ   280тЖТ    if (!exam.trim()) { toast({ title: 'Enter exam name', description: 'e.g. Chapter 1 Test', variant: 'destructive' }); return; }
   281тЖТ   281тЖТ    if (!courseId) { toast({ title: 'Select a course first', variant: 'destructive' }); return; }
   282тЖТ   282тЖТ    if (students.length === 0) { toast({ title: 'No students', variant: 'destructive' }); return; }
   283тЖТ   283тЖТ    setSaving(true);
   284тЖТ   284тЖТ    try {
   285тЖТ   285тЖТ      const records = students.map(s => {

   394тЖТ   394тЖТfunction MaterialUploadForm({ classId, courseId, teacherId, onSaved }: { classId: string; courseId: string; teacherId: string; onSaved: () => void }) {
   395тЖТ   395тЖТ  const [title, setTitle] = useState('');
   396тЖТ   396тЖТ  const [description, setDescription] = useState('');
   397тЖТ   397тЖТ  const [mode, setMode] = useState<'file' | 'link'>('file');
   398тЖТ   398тЖТ  const [linkUrl, setLinkUrl] = useState('');
   399тЖТ   399тЖТ  const [file, setFile] = useState<File | null>(null);
   400тЖТ   400тЖТ  const [saving, setSaving] = useState(false);
   401тЖТ   401тЖТ
   402тЖТ   402тЖТ  const handleFile = (f: File | null) => {
   403тЖТ   403тЖТ    if (!f) { setFile(null); return; }
   404тЖТ   404тЖТ    const allowed = ['application/pdf', 'application/vnd.openxmlformats-officedocument.wordprocessingml.document',
   405тЖТ   405тЖТ      'application/vnd.openxmlformats-officedocument.presentationml.presentation', 'image/png', 'image/jpeg'];
   406тЖТ   406тЖТ    const ext = f.name.split('.').pop()?.toLowerCase();
   407тЖТ   407тЖТ    const okExt = ['pdf', 'docx', 'ppt', 'pptx', 'png', 'jpg', 'jpeg'].includes(ext || '');
   408тЖТ   408тЖТ    if (!allowed.includes(f.type) && !okExt) {
   409тЖТ   409тЖТ      toast({ title: 'Unsupported file type', description: 'Use PDF, DOCX, PPT, PNG, or JPG.', variant: 'destructive' });
   410тЖТ   410тЖТ      return;
   411тЖТ   411тЖТ    }
   412тЖТ   412тЖТ    if (f.size > 8 * 1024 * 1024) {
   413тЖТ   413тЖТ      toast({ title: 'File too large', description: 'Max 8 MB.', variant: 'destructive' });
   414тЖТ   414тЖТ      return;
   415тЖТ   415тЖТ    }
   416тЖТ   416тЖТ    setFile(f);
   417тЖТ   417тЖТ  };
   418тЖТ   418тЖТ
   419тЖТ   419тЖТ  const submit = async () => {
   420тЖТ   420тЖТ    if (!title.trim()) { toast({ title: 'Title required', variant: 'destructive' }); return; }
   421тЖТ   421тЖТ    if (mode === 'file' && !file) { toast({ title: 'Choose a file', variant: 'destructive' }); return; }
   422тЖТ   422тЖТ    if (mode === 'link' && !linkUrl.trim()) { toast({ title: 'Link URL required', variant: 'destructive' }); return; }
   423тЖТ   423тЖТ
   424тЖТ   424тЖТ    setSaving(true);
   425тЖТ   425тЖТ    try {
   426тЖТ   426тЖТ      let body: any = { classId, courseId, title, description, teacherId };
   427тЖТ   427тЖТ      if (mode === 'file' && file) {
   428тЖТ   428тЖТ        const base64 = await new Promise<string>((resolve, reject) => {
   429тЖТ   429тЖТ          const reader = new FileReader();
   430тЖТ   430тЖТ          reader.onload = () => {
   431тЖТ   431тЖТ            const result = reader.result as string;
   432тЖТ   432тЖТ            // strip "data:<mime>;base64," prefix
   433тЖТ   433тЖТ            const idx = result.indexOf(',');
   434тЖТ   434тЖТ            resolve(idx >= 0 ? result.slice(idx + 1) : result);
   435тЖТ   435тЖТ          };
   436тЖТ   436тЖТ          reader.onerror = reject;
   437тЖТ   437тЖТ          reader.readAsDataURL(file);
   438тЖТ   438тЖТ        });
   439тЖТ   439тЖТ        body = { ...body, fileType: file.type || 'application/octet-stream', fileName: file.name, fileData: base64 };
   440тЖТ   440тЖТ      } else if (mode === 'link') {
   441тЖТ   441тЖТ        body = { ...body, linkUrl };
   442тЖТ   442тЖТ      }
   443тЖТ   443тЖТ      await api.addCourseMaterial(body);
   444тЖТ   444тЖТ      toast({ title: 'Material uploaded!', description: 'Students can now access it.' });
   445тЖТ   445тЖТ      setTitle(''); setDescription(''); setLinkUrl(''); setFile(null);
   446тЖТ   446тЖТ      onSaved();
   447тЖТ   447тЖТ    } catch (e: any) { toast({ title: 'Upload failed', description: e.message, variant: 'destructive' }); }
   448тЖТ   448тЖТ    finally { setSaving(false); }
   449тЖТ   449тЖТ  };
   450тЖТ   450тЖТ
   451тЖТ   451тЖТ  return (
   452тЖТ   452тЖТ    <Card className="p-5 space-y-3">
   453тЖТ   453тЖТ      <div className="flex items-center gap-2 mb-1">
   454тЖТ   454тЖТ        <div className="h-8 w-8 rounded-lg bg-accent0/15 grid place-items-center"><Paperclip className="h-4 w-4 text-primary" /></div>
   455тЖТ   455тЖТ        <h3 className="font-bold text-sm">New Course Material</h3>
   456тЖТ   456тЖТ      </div>
   457тЖТ   457тЖТ      <div className="flex gap-1 p-1 rounded-lg bg-muted/60 w-fit">
   458тЖТ   458тЖТ        <button onClick={() => setMode('file')} className={`px-3 py-1.5 rounded-md text-xs font-medium transition ${mode === 'file' ? 'bg-card shadow-sm' : 'text-muted-foreground'}`}>
   459тЖТ   459тЖТ          <FileText className="h-3.5 w-3.5 inline mr-1" /> File
   460тЖТ   460тЖТ        </button>
   461тЖТ   461тЖТ        <button onClick={() => setMode('link')} className={`px-3 py-1.5 rounded-md text-xs font-medium transition ${mode === 'link' ? 'bg-card shadow-sm' : 'text-muted-foreground'}`}>
   462тЖТ   462тЖТ          <Link2 className="h-3.5 w-3.5 inline mr-1" /> Link
   463тЖТ   463тЖТ        </button>
   464тЖТ   464тЖТ      </div>
   465тЖТ   465тЖТ      <div><Label className="text-xs">Title *</Label><Input value={title} onChange={e => setTitle(e.target.value)} placeholder="Chapter 5 тАФ Lecture Notes" className="mt-1" /></div>
   466тЖТ   466тЖТ      <div><Label className="text-xs">Description</Label><textarea value={description} onChange={e => setDescription(e.target.value)} rows={2} placeholder="Brief description of the materialтАж" className="w-full mt-1 rounded-md border border-border bg-card p-2 text-sm resize-none" /></div>
   467тЖТ   467тЖТ      {mode === 'file' ? (
   468тЖТ   468тЖТ        <div>
   469тЖТ   469тЖТ          <Label className="text-xs">File (PDF, DOCX, PPT, PNG, JPG тАФ max 8 MB)</Label>
   470тЖТ   470тЖТ          <input type="file" accept=".pdf,.docx,.ppt,.pptx,.png,.jpg,.jpeg" onChange={e => handleFile(e.target.files?.[0] || null)} className="block w-full mt-1 text-sm file:mr-3 file:py-1.5 file:px-3 file:rounded-md file:border-0 file:bg-accent0/15 file:text-primary hover:file:bg-accent0/25 file:font-medium file:text-sm cursor-pointer" />
   471тЖТ   471тЖТ          {file && <div className="mt-2 text-xs text-muted-foreground flex items-center gap-1.5"><Paperclip className="h-3 w-3" /> {file.name} ┬╖ {(file.size / 1024).toFixed(0)} KB</div>}
   472тЖТ   472тЖТ        </div>
   473тЖТ   473тЖТ      ) : (
   474тЖТ   474тЖТ        <div><Label className="text-xs">Link URL *</Label><Input value={linkUrl} onChange={e => setLinkUrl(e.target.value)} placeholder="https://docs.google.com/тАж" className="mt-1" /></div>
   475тЖТ   475тЖТ      )}
   476тЖТ   476тЖТ      <Button size="sm" className="bg-primary hover:bg-primary/90 text-white" disabled={saving} onClick={submit}>
   477тЖТ   477тЖТ        {saving ? <><Loader2 className="h-4 w-4 mr-1.5 animate-spin" /> UploadingтАж</> : 'Upload'}
   478тЖТ   478тЖТ      </Button>
   479тЖТ   479тЖТ    </Card>
   480тЖТ   480тЖТ  );
   481тЖТ   481тЖТ}

   483тЖТ   483тЖТfunction MaterialCard({ material, teacherView = false }: { material: MaterialItem; teacherView?: boolean }) {
