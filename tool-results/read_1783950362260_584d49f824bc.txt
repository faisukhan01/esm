    43→export function StudentPortal({ activeModule, user }: { activeModule: string; user: any }) {
    44→  const [attendance, setAttendance] = useState<any>(null);
    45→  const [results, setResults] = useState<any>(null);
    46→  const [diary, setDiary] = useState<any[]>([]);
    47→  const [courses, setCourses] = useState<Course[]>([]);
    48→  const [classId, setClassId] = useState<string>('');
    49→  const [announcements, setAnnouncements] = useState<Announcement[]>([]);
    50→  const [selectedCourse, setSelectedCourse] = useState<{ course: Course; initialTab: 'materials' | 'results' | 'attendance' } | null>(null);
    51→
    52→  const refresh = () => {
    53→    if (user?.id) {
    54→      api.getAttendance({ studentId: user.id }).then(setAttendance).catch(() => {});
    55→      api.getResults({ studentId: user.id }).then(setResults).catch(() => {});
    56→      if (user?.branchId) api.getDiary({ branchId: user.branchId }).then(setDiary).catch(() => {});
    57→      api.getStudentCourses().then(setCourses).catch(() => setCourses([]));
    58→      api.getAnnouncements().then(setAnnouncements).catch(() => setAnnouncements([]));
    59→    }
    60→  };
    61→
    62→  // Resolve the student's classId from the branch's classes by matching user.class (class name)
    63→  useEffect(() => {
    64→    if (user?.branchId) {
    65→      api.getClasses(user.branchId)
    66→        .then((cls: any[]) => {
    67→          const match = cls.find(c => c.name === user.class);
    68→          setClassId(match?.id || '');
    69→        })
    70→        .catch(() => setClassId(''));
    71→    }
    72→  }, [user?.branchId, user?.class]);
    73→
    74→  useEffect(() => { refresh(); }, [user?.id, user?.branchId]);
    75→
    76→  if (activeModule === 'student-overview' && selectedCourse) {
    77→    return (
    78→      <CourseDetail
    79→        course={selectedCourse.course}
    80→        classId={classId}
    81→        studentId={user?.id}
    82→        initialTab={selectedCourse.initialTab}
    83→        onBack={() => setSelectedCourse(null)}
    84→      />
    85→    );
    86→  }
    87→
    88→  if (activeModule === 'my-attendance') return <MyAttendance attendance={attendance} />;
    89→  if (activeModule === 'my-results') return <MyResults results={results} />;
    90→  if (activeModule === 'my-timetable') return <MyTimetable />;
    91→  if (activeModule === 'my-diary') return <MyDiary diary={diary} />;
    92→  if (activeModule === 'my-announcements') return <MyAnnouncements announcements={announcements} loading={false} />;
    93→  if (activeModule === 'my-invoices') return <MyInvoices user={user} />;
    94→  return (
    95→    <StudentOverview
    96→      user={user}
    97→      attendance={attendance}
    98→      results={results}
    99→      courses={courses}
   100→      announcements={announcements}
   101→      onOpenCourse={(course, tab) => setSelectedCourse({ course, initialTab: tab })}
   102→    />
   103→  );
   104→}

   282→function CourseMaterialsView({ classId, courseId }: { classId: string; courseId: string }) {
   283→  const [materials, setMaterials] = useState<MaterialItem[]>([]);
   284→  const [loading, setLoading] = useState(true);
   285→
   286→  useEffect(() => {
   287→    if (!classId || !courseId) return;
   288→    api.getCourseMaterials({ classId, courseId })
   289→      .then(setMaterials)
   290→      .catch(() => setMaterials([]))
   291→      .finally(() => setLoading(false));
   292→  }, [classId, courseId]);
   293→
   294→  if (!classId) {
   295→    return <EmptyState icon={FileText} title="Class not found" desc="We couldn't resolve your class. Please contact your Branch Manager." />;
   296→  }
   297→
   298→  if (loading) {
   299→    return <Card className="p-8 text-center"><Loader2 className="h-6 w-6 animate-spin mx-auto text-muted-foreground" /></Card>;
   300→  }
   301→
   302→  if (materials.length === 0) {
   303→    return <EmptyState icon={FileText} title="No materials yet" desc="Your teacher hasn't uploaded any materials for this course yet. Check back later." />;
   304→  }
   305→
   306→  return (
   307→    <div className="space-y-4">
   308→      <p className="text-sm text-muted-foreground">{materials.length} material{materials.length === 1 ? '' : 's'} available</p>
   309→      <div className="grid sm:grid-cols-2 gap-3">
   310→        {materials.map(m => <MaterialCard key={m.id} material={m} />)}
   311→      </div>
   312→    </div>
   313→  );
   314→}

   316→function MaterialCard({ material }: { material: MaterialItem }) {
   317→  const [downloading, setDownloading] = useState(false);
   318→  const isLink = !!material.linkUrl;
   319→  const Icon = isLink ? Link2 : FileText;
   320→  const ext = material.fileName?.split('.').pop()?.toUpperCase() || (isLink ? 'LINK' : 'FILE');
   321→  const createdAt = material.createdAt ? new Date(material.createdAt).toLocaleDateString() : '';
   322→
   323→  const handleDownload = async () => {
   324→    if (isLink && material.linkUrl) {
   325→      window.open(material.linkUrl, '_blank', 'noopener,noreferrer');
   326→      return;
   327→    }
   328→    setDownloading(true);
   329→    try {
   330→      const res = await api.downloadMaterialBlob(material.id);
   331→      if ('linkUrl' in res) {
   332→        window.open(res.linkUrl, '_blank', 'noopener,noreferrer');
   333→      } else {
   334→        const url = URL.createObjectURL(res.blob);
   335→        const a = document.createElement('a');
   336→        a.href = url;
   337→        a.download = res.fileName || material.fileName || 'download';
   338→        document.body.appendChild(a);
   339→        a.click();
   340→        document.body.removeChild(a);
   341→        URL.revokeObjectURL(url);
   342→      }
   343→    } catch (e: any) {
   344→      toast({ title: 'Download failed', description: e.message, variant: 'destructive' });
   345→    } finally { setDownloading(false); }
   346→  };
   347→
   348→  return (
   349→    <Card className="p-4 hover:shadow-md transition">
   350→      <div className="flex items-start gap-3">
   351→        <div className={`h-10 w-10 rounded-lg grid place-items-center shrink-0 ${isLink ? 'bg-[oklch(0.95_0.01_260)]0/15' : 'bg-[oklch(0.95_0.01_260)]0/15'}`}>
   352→          <Icon className={`h-5 w-5 ${isLink ? 'text-[oklch(0.22_0.04_260)]' : 'text-[oklch(0.22_0.04_260)]'}`} />
   353→        </div>
   354→        <div className="flex-1 min-w-0">
   355→          <div className="flex items-center gap-2 mb-0.5">
   356→            <h4 className="font-medium text-sm truncate">{material.title}</h4>
   357→            <Badge variant="outline" className="text-[10px] font-mono shrink-0">{ext}</Badge>
   358→          </div>
   359→          {material.description && <p className="text-xs text-muted-foreground line-clamp-2">{material.description}</p>}
   360→          <div className="text-[11px] text-muted-foreground mt-1">{createdAt}{isLink ? ' · External link' : ''}</div>
   361→        </div>
   362→      </div>
   363→      <div className="mt-3 flex justify-end">
   364→        <Button size="sm" variant="outline" onClick={handleDownload} disabled={downloading}>
   365→          {downloading ? <Loader2 className="h-3.5 w-3.5 mr-1.5 animate-spin" /> : <Download className="h-3.5 w-3.5 mr-1.5" />}
   366→          {isLink ? 'Open Link' : 'Download'}
   367→        </Button>
   368→      </div>
   369→    </Card>
   370→  );
   371→}

   373→function CourseResultsView({ courseId, studentId }: { courseId: string; studentId: string }) {
   374→  const [results, setResults] = useState<any[]>([]);
   375→  const [loading, setLoading] = useState(true);
   376→
   377→  useEffect(() => {
   378→    api.getResults({ studentId })
   379→      .then((r: any) => {
   380→        const entries = r?.entries || [];
   381→        // Filter to this course; backend may not return courseId on student-scoped results, so include all if courseId missing
   382→        const filtered = entries.filter((e: any) => !e.courseId || e.courseId === courseId);
   383→        setResults(filtered.length > 0 ? filtered : entries);
   384→      })
   385→      .catch(() => setResults([]))
   386→      .finally(() => setLoading(false));
   387→  }, [courseId, studentId]);
   388→
   389→  if (loading) return <Card className="p-8 text-center"><Loader2 className="h-6 w-6 animate-spin mx-auto text-muted-foreground" /></Card>;
   390→  if (results.length === 0) {
   391→    return <EmptyState icon={GraduationCap} title="No results yet" desc="Your teacher hasn't posted any results for this course yet." />;
   392→  }
   393→
   394→  return (
   395→    <Card className="p-5">
   396→      <div className="space-y-3">
   397→        {results.map((r: any, idx: number) => (
   398→          <div key={r.id || idx} className="flex items-center gap-3">
   399→            <div className="h-9 w-9 rounded-lg bg-[oklch(0.95_0.01_260)]0/15 grid place-items-center shrink-0"><BookOpen className="h-4 w-4 text-[oklch(0.22_0.04_260)]" /></div>
   400→            <div className="flex-1">
   401→              <div className="flex items-center justify-between mb-1"><span className="font-medium text-sm">{r.exam || 'Exam'}</span><span className="font-bold text-sm">{r.marks}/{r.totalMarks}</span></div>
   402→              <div className="h-2 rounded-full bg-muted overflow-hidden"><div className="h-full bg-[oklch(0.22_0.04_260)]" style={{ width: `${r.percentage || (r.totalMarks ? (r.marks / r.totalMarks) * 100 : 0)}%` }} /></div>
   403→              <div className="text-[11px] text-muted-foreground mt-1">{r.date}</div>
   404→            </div>
   405→            <Badge variant="outline" className="font-bold">{r.grade}</Badge>
   406→          </div>
   407→        ))}
   408→      </div>
   409→    </Card>
   410→  );
   411→}

   664→// Print the challan using a hidden iframe — no new tab is opened.
   665→// The browser's print dialog lets the student pick "Save as PDF" to download the file.
   666→function printChallanInIframe(html: string) {
   667→  // Reuse an existing hidden iframe if present, otherwise create one.
   668→  let iframe = document.getElementById('esm-challan-frame') as HTMLIFrameElement | null;
   669→  if (!iframe) {
   670→    iframe = document.createElement('iframe');
   671→    iframe.id = 'esm-challan-frame';
   672→    iframe.style.position = 'fixed';
   673→    iframe.style.right = '0';
   674→    iframe.style.bottom = '0';
   675→    iframe.style.width = '0';
   676→    iframe.style.height = '0';
   677→    iframe.style.border = '0';
   678→    iframe.setAttribute('aria-hidden', 'true');
   679→    iframe.setAttribute('title', 'ESM challan print frame');
   680→    document.body.appendChild(iframe);
   681→  }
   682→
   683→  const doc = iframe.contentWindow?.document;
   684→  if (!doc) {
   685→    toast({ title: 'Print unavailable', description: 'Your browser blocked the print frame. Try the popup version instead.', variant: 'destructive' });
   686→    return;
   687→  }
   688→
   689→  // Replace the iframe's document with the fresh challan HTML.
   690→  doc.open();
   691→  doc.write(html);
   692→  doc.close();
   693→
   694→  // Defer the print call so the iframe has a tick to lay out its content.
   695→  // We avoid relying on iframe.onload because reusing the same iframe means
   696→  // the load event already fired on a prior call and won't fire again.
   697→  const win = iframe.contentWindow;
   698→  setTimeout(() => {
   699→    try { win?.focus(); win?.print(); }
   700→    catch {
   701→      toast({ title: 'Print failed', description: 'Please allow popups / printing for this site.', variant: 'destructive' });
   702→    }
   703→  }, 300);
   704→}

   707→  const html = buildChallanHTML(challan, instituteName);
   708→  printChallanInIframe(html);
